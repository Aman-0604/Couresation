create database book;
use book; 
create table goal(goal_id int primary key AUTO_INCREMENT, goal_name varchar(200) not null);
create table course(course_id varchar(5) primary key, course_name varchar(100), course_description varchar(2000), goal_id int, foreign key(goal_id) references goal(goal_id));
create table student(student_id int primary key AUTO_INCREMENT, name varchar(200) not null, address varchar(200), phone_number int, email varchar(200) not null, password varchar(200) not null, goal_id int, CHECK (email REGEXP '^[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Za-z]{2,}$'));
create table teacher(teacher_id int primary key AUTO_INCREMENT, name varchar(200) not null, address varchar(200), phone_number int, email varchar(200) not null, password varchar(200) not null, course_id int, CHECK (email REGEXP '^[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Za-z]{2,}$'));
create table video_link(link varchar(500) primary key,title varchar(500), course_id varchar(5), FOREIGN KEY (course_id) REFERENCES course(course_id));
